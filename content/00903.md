```bash
git clone https://github.com/eugr/spark-vllm-docker.git
cd spark-vllm-docker
./build-and-copy.sh
```

[`sakamakismile/Qwen3.6-27B-Text-NVFP4-MTP`](https://huggingface.co/sakamakismile/Qwen3.6-27B-Text-NVFP4-MTP)

`vllm-qwen3.6-27b.sh`

```bash
vllm serve \
  sakamakismile/Qwen3.6-27B-Text-NVFP4-MTP \
  --language-model-only \
  --enable-auto-tool-choice \
  --tool-call-parser qwen3_coder \
  --reasoning-parser qwen3 \
  --gpu-memory-utilization 0.6 \
  --quantization modelopt \
  --kv-cache-dtype fp8 \
  --max-num-seqs 4 \
  --host 0.0.0.0 \
  --port 8000 \
  --load-format fastsafetensors \
  --attention-backend flashinfer \
  --default-chat-template-kwargs '{"enable_thinking":false}' \
  --enable-prefix-caching \
  --speculative-config '{"method":"qwen3_5_mtp","num_speculative_tokens":2}'
```

```bash
./launch-cluster.sh --solo -d --apply-mod mods/drop-caches --launch-script vllm-qwen3.6-27b.sh
```

docker logs -f vllm_node
