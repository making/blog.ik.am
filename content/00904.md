[ds4(DwarfStar 4)](https://github.com/antirez/ds4)がDGX Sparkに対応したので、試してみます。

```bash
sudo mkdir -p /opt/ds4
sudo chown -R $USER:$USER /opt/ds4
```

```bash
git clone https://github.com/antirez/ds4.git .
```

```bash
$ git rev-parse --short HEAD
```

```bash
./download_model.sh q2
```


```bash
cat <<EOF | sudo tee /etc/systemd/system/ds4-server.service > /dev/null
[Unit]
Description=DwarfStar 4 Server
Documentation=https://github.com/antirez/ds4
After=network-online.target
Wants=network-online.target

[Service]
User=$USER
Group=$USER
Type=simple
ExecStart=/opt/ds4/ds4-server --ctx 200000 --kv-disk-dir /tmp/ds4-kv --kv-disk-space-mb 8192 --host 0.0.0.0

# Resource control
Restart=on-failure
RestartSec=30s
StartLimitInterval=300
StartLimitBurst=5

[Install]
WantedBy=multi-user.target
EOF
```

```bash
sudo systemctl daemon-reload
sudo systemctl enable ds4-server
sudo systemctl start ds4-server
sudo systemctl status ds4-server
```
